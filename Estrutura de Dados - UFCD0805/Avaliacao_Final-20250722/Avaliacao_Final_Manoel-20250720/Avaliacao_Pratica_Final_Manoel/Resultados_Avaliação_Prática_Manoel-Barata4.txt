// ===== PROCEDIMENTOS =====


PROCEDIMENTO criar_no(valor): NÓ_ÁRVORE
INÍCIO
    novo_no: NÓ_ÁRVORE
    novo_no.dados = valor
    novo_no.esquerdo = NULL
    novo_no.direito = NULL
    RETORNAR novo_no
FIM


PROCEDIMENTO pre_ordem(no: PONTEIRO para NÓ_ÁRVORE)
INÍCIO
    SE (no ≠ NULL) ENTÃO
        ESCREVER no.dados, ", "
        pre_ordem(no.esquerdo)
        pre_ordem(no.direito)
    FIM SE
FIM

    
 PROCEDIMENTO em_ordem_nums(no: PONTEIRO para NÓ_ÁRVORE)
INÍCIO
    SE (no ≠ NULL) ENTÃO
        em_ordem_nums(no.esquerdo)
        ESCREVER no.dados, ", "
        em_ordem_nums(no.direito)
    FIM SE
FIM


FUNÇÃO calcular_altura(no: PONTEIRO para NÓ_ÁRVORE): INTEIRO
INÍCIO
    SE (no = NULL) ENTÃO
        RETORNAR -1  // Convenção: árvore vazia tem altura -1
    FIM SE
    
    altura_esquerda = calcular_altura(no.esquerdo)
    altura_direita = calcular_altura(no.direito)
    
    SE (altura_esquerda > altura_direita) ENTÃO
        RETORNAR altura_esquerda + 1
    SENÃO
        RETORNAR altura_direita + 1
    FIM SE
FIM
