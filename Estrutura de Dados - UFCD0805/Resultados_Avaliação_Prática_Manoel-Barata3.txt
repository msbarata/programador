contador: INTEIRO


// ===== PROCEDIMENTOS =====


PROCEDIMENTO push(url: TEXTO)
INÍCIO
    SE (topo = MAX_HIST-1) ENTÃO
        ESCREVER "Pilha cheia!"

    topo = topo + 1
    pilha[topo] = url
    ESCREVER "Visitando: ", url
FIM



FUNÇÃO peek(): TEXTO
INÍCIO
    SE (topo = -1) ENTÃO
        RETORNAR ""
    SENÃO
        RETORNAR pilha[topo]
    FIM SE
FIM


FUNÇÃO pop(): TEXTO

FIM
